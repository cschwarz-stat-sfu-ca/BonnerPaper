#  estimate N* =- sum(U*) + n

    ############################################################
model {
    
# input data are n (number of samples), x_i=p_i

  # generate the HT equivalent estimator
  for(i in 1:n){
     m.star[i] ~ dnegbin(x[i],1)
  }
  U.star= sum(m.star)
  N.star= U.star + n
}
    ") # End of the model
